CREATE OR REPLACE FUNCTION hrmanagement_store_hr_info(
    in_companyname varchar,
    in_companywebsite varchar,
    in_hrname varchar,
    in_hremail varchar,
    in_hrmobile varchar,
    in_positionname varchar
)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
    var_position_id integer;
BEGIN

    INSERT INTO position_master(position_name)
    VALUES (in_positionname)
    ON CONFLICT (position_name)
    DO UPDATE SET position_name = EXCLUDED.position_name
    RETURNING id INTO var_position_id;

    INSERT INTO hr_info(
        company_name,
        hr_name,
        email,
        mobileno,
        company_website,
        position_id
    )
    VALUES (
        in_companyname,
        in_hrname,
        in_hremail,
        in_hrmobile,
        in_companywebsite,
        var_position_id
    );

END;
$$;

CREATE OR REPLACE FUNCTION public.hrmanagement_get_position_list(

    RETURNS TABLE(id integer, position_name character varying)

$BODY$;

CREATE OR REPLACE FUNCTION public.hrmanagement_get_hr_info_list(
	in_searchterm character varying,
	in_filtername character varying)
    RETURNS TABLE(id integer, company_name character varying, hr_name character varying, position_name character varying, is_applied boolean, is_verified boolean) 
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE SECURITY DEFINER PARALLEL UNSAFE
    ROWS 1000

AS $BODY$
DECLARE SQL VARCHAR;
BEGIN

	SQL := 'select
	hr.id,
	hr.company_name,
	hr.hr_name,
	pm.position_name,
	hr.is_applied,
	hr.is_verified
	from hr_info hr
	join position_master pm on pm.id = hr.position_id
	WHERE 1 = 1';

	IF in_searchterm IS NOT NULL THEN
	SQL := SQL || ' AND (hr.company_name ilike ' || quote_literal(in_searchterm || '%') || ' OR hr.hr_name ilike ' ||
	quote_literal(in_searchterm || '%') || ' )';
	END IF;

	IF in_filtername <> 'all_records' THEN
		IF in_filtername = 'not_applied_yet' THEN
			SQL := SQL || ' AND hr.is_applied = false';
		ELSIF in_filtername = 'hr_desc' THEN
			SQL := SQL || ' ORDER BY hr.hr_name DESC';
		ELSE SQL := SQL || 'ORDER BY hr.hr_name';
		END IF;
	END IF;
		

	RETURN QUERY EXECUTE SQL;
	

END;
$BODY$;

CREATE OR REPLACE FUNCTION public.hrmanagement_store_bulk_template_info(
	in_eventid integer,
	in_positionname character varying,
	in_templateid integer,
	in_resumepath character varying,
	in_selected_hr_ids jsonb,
	in_totalcount integer)
    RETURNS TABLE(campaign_id integer, notification_id integer) 
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE SECURITY DEFINER PARALLEL UNSAFE
    ROWS 1000

AS $BODY$
DECLARE
    v_campaign_id integer;
    v_notification_id integer;
BEGIN

    -- 1️⃣ Insert campaign and capture ID
    INSERT INTO email_campaigns (
        event_id,
        position_name,
        template_id,
        resume_path,
        selected_hr_ids,
        total_count,
        status
    )
    VALUES (
        in_eventid,
        in_positionname,
        in_templateid,
        in_resumepath,
        in_selected_hr_ids,
        in_totalcount,
        'PROCESSING'
    )
    RETURNING email_campaigns.campaign_id
    INTO v_campaign_id;

    -- 2️⃣ Insert notification using captured campaign_id
    INSERT INTO notifications (
        campaign_id,
        notification_message,
        is_read
    )
    VALUES (
        v_campaign_id,
        'Start Bulk Email Send.',
        false
    )
    RETURNING notifications.notification_id
    INTO v_notification_id;

    -- 3️⃣ Return both values
    RETURN QUERY
    SELECT v_campaign_id, v_notification_id;

END;
$BODY$;

CREATE OR REPLACE FUNCTION public.email_campaign_get_job_info(
	in_hrid integer,
	in_campaignid integer,
	in_templateid integer,
	in_sendcount integer)
    RETURNS jsonb
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE SECURITY DEFINER PARALLEL UNSAFE
AS $BODY$
DECLARE
    hr_data JSONB;
    template_data JSONB;
BEGIN

    -- 1️⃣ Update JSONB status to 'processing'
    UPDATE email_campaigns
    SET selected_hr_ids = (
        SELECT jsonb_agg(
            CASE 
                WHEN (elem->>'hr_id')::INT = in_hrid
                THEN jsonb_set(elem, '{status}', '"processing"')
                ELSE elem
            END
        )
        FROM jsonb_array_elements(selected_hr_ids) elem
    ), sent_count = in_sendCount
    WHERE campaign_id = in_campaignid;

    -- 2️⃣ Get HR info as JSON
    SELECT to_jsonb(h)
    INTO hr_data
    FROM (
        SELECT company_name, hr_name, email
        FROM hr_info
        WHERE id = in_hrid
    ) h;

    -- 3️⃣ Get Template info as JSON
    SELECT to_jsonb(t)
    INTO template_data
    FROM (
        SELECT subject_title, body
        FROM template_master
        WHERE id = in_templateid
    ) t;

    -- 4️⃣ Return combined JSON array
    RETURN jsonb_build_array(
        jsonb_build_object('hr_info', hr_data),
        jsonb_build_object('template_info', template_data)
    );

END;
$BODY$;

CREATE OR REPLACE FUNCTION public.email_campaign_complete_email_campaign_job(
    in_campaignid integer)
RETURNS void
LANGUAGE plpgsql
COST 100
VOLATILE SECURITY DEFINER PARALLEL UNSAFE
AS $BODY$
DECLARE
    has_incomplete boolean;
BEGIN

    -- Check if any HR status is not completed
    SELECT EXISTS (
        SELECT 1
        FROM email_campaigns,
             jsonb_array_elements(selected_hr_ids) elem
        WHERE campaign_id = in_campaignid
        AND elem->>'status' <> 'completed'
    ) INTO has_incomplete;

    IF has_incomplete THEN
        -- If any HR is not completed
        UPDATE email_campaigns
        SET status = 'ERROR'
        WHERE campaign_id = in_campaignid;

    ELSE
        -- If all HR are completed
        UPDATE email_campaigns
        SET 
        status = 'COMPLETED'
        WHERE campaign_id = in_campaignid;

    END IF;

END;
$BODY$;

CREATE OR REPLACE FUNCTION email_campaign_update_hr_status(
    in_campaignid integer,
    in_hrid integer,
    in_status varchar
)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN

-- Update email_campaigns JSON
UPDATE email_campaigns
SET selected_hr_ids = (
    SELECT jsonb_agg(
        CASE 
            WHEN (elem->>'hr_id')::INT = in_hrid
            THEN jsonb_set(elem, '{status}', to_jsonb(in_status))
            ELSE elem
        END
    )
    FROM jsonb_array_elements(selected_hr_ids) elem
)
WHERE campaign_id = in_campaignid;

-- Store logs of application if the successfully send email
insert into application_history(hr_id, status) values (in_hrid, in_status);

-- If email successfully sent
IF in_status = 'completed' THEN

    UPDATE hr_info
    SET is_applied = true,
        is_verified = true
    WHERE id = in_hrid;

    INSERT INTO application_history(hr_id)
    VALUES (in_hrid);

-- If email failed
ELSE

    UPDATE hr_info
    SET is_applied = true,
        is_verified = false
    WHERE id = in_hrid;

END IF;

END;
$$;

CREATE OR REPLACE FUNCTION email_jobs_get_jobs_list()
RETURNS TABLE(
	jobid integer,
	job_status varchar,
	job_event varchar,
	created text
)
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN

-- select * from email_jobs_get_jobs_list()

RETURN QUERY
SELECT 
	ec.campaign_id AS jobid,
	ec.status AS job_status,
	em.event_type AS job_event,
	to_char(ec.created_at, 'DD/MM/YYYY') AS created
FROM email_campaigns ec
JOIN event_master em 
	ON em.event_id = ec.event_id;

END;
$$;

CREATE OR REPLACE FUNCTION email_campaign_get_hr_list(
    in_campaign_id integer
)
RETURNS TABLE (
    hr_id integer,
    hr_name varchar,
    company varchar,
    email varchar,
    status varchar,
    created_at timestamp
)
LANGUAGE sql
SECURITY DEFINER
AS $$
SELECT
    hi.id AS hr_id,
    hi.hr_name AS hr_name,
    hi.company_name,
    hi.email,
    elem->>'status' AS status,
    ec.created_at
FROM email_campaigns ec
JOIN jsonb_array_elements(ec.selected_hr_ids) elem ON true
JOIN hr_info hi 
    ON hi.id = (elem->>'hr_id')::int
WHERE ec.campaign_id = in_campaign_id;
$$;

CREATE OR REPLACE FUNCTION template_config_update_template_info(
    in_templateid integer,
    in_template_subject varchar,
    in_template_name varchar,
    in_template_body varchar
)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN

    -- Store old template data in log table
    INSERT INTO template_updation_log(
        template_id,
        template_name,
        template_subject,
        template_body
    )
    SELECT 
        id,
        template_name,
        subject_title,
        body
    FROM template_master
    WHERE id = in_templateid;

    -- Update subject
    IF in_template_subject IS NOT NULL THEN
        UPDATE template_master
        SET subject_title = in_template_subject
        WHERE id = in_templateid;
    END IF;

    -- Update name
    IF in_template_name IS NOT NULL THEN
        UPDATE template_master
        SET template_name = in_template_name
        WHERE id = in_templateid;
    END IF;

    -- Update body
    IF in_template_body IS NOT NULL THEN
        UPDATE template_master
        SET body = in_template_body
        WHERE id = in_templateid;
    END IF;

END;
$$;
